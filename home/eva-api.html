Python
from openquant import EvaClient

client = EvaClient(api_key="aq_live_...")

result = client.evaluate(
    thesis="Long NVDA on AI capex cycle, 6mo horizon",
    portfolio_context={"sector_weight": "tech_35pct"}
)

# result.sharpe_estimate  → 1.82
# result.confidence       → 0.74
# result.risk_flags       → ["sector_concentration"]
